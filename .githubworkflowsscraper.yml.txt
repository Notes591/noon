name: Noon Scraper -> Google Sheets

on:
  schedule:
    - cron: '*/30 * * * *'   # every 30 minutes (عدل لو تحب)
  workflow_dispatch: {}      # allow manual trigger

jobs:
  run-scraper:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout repo
        uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.11'

      - name: Install dependencies
        run: pip install -r requirements.txt

      - name: Create service account file from secret
        env:
          GCP_SA_KEY: ${{ secrets.GCP_SA_KEY }}
        run: |
          echo "$GCP_SA_KEY" > service-account.json
          chmod 600 service-account.json

      - name: Run scraper (single pass)
        env:
          # ensure the env var points to the file name used in scraper.py
          SERVICE_ACCOUNT_JSON: service-account.json
        run: |
          python scraper.py
